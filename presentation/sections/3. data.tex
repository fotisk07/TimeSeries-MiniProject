\section{Data Analysis}

\begin{frame}{Data Sources \& Challenges}
    \frametitle{High-Frequency vs. Sparse Event Data}
    
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{1. Tracking Data (GPS)}
            \begin{itemize}
                \item \textbf{Source:} Fitogether (10Hz) \cite{Kim2022SoccerCPD}, SkillCorner (10Hz) \cite{SkillCorner2020OpenData}.
                \item \textbf{Quality:} Dense, clean signals.
                \item \textbf{Prep:} Normalized to centered metric pitch.
            \end{itemize}
        \end{column}
        \begin{column}{0.5\textwidth}
            \textbf{2. Event Data (StatsBomb)}
            \begin{itemize}
                \item \textbf{Source:} On-ball events only.
                \item \textbf{Challenge:} Extreme sparsity ($<10$ pts/min vs 600+ for GPS).
                \item \textbf{Bias:} Spatial bias toward the ball.
            \end{itemize}
        \end{column}
    \end{columns}
    
    \vspace{0.3cm}
    
    \textbf{Novel Adaptation for Event Data:}
    \begin{itemize}
        \item \textbf{Pseudo-Trajectories:} Aggregated events over 5-minute rolling windows.
        \item \textbf{Proxy Graph:} Average adjacency matrices instead of frame-by-frame Delaunay.
    \end{itemize}
\end{frame}

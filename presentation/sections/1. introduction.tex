\section{Introduction}
\subsection{Context and Problem Statement}
\begin{frame}{Context \& Problem Statement}
    \frametitle{Context: The Fluidity of Soccer}
    \begin{columns}
        \begin{column}{0.6\textwidth}
            \textbf{The Challenge in Sports Analytics}
            \begin{itemize}
                \item \textbf{Dynamic Formations:} Teams constantly reshape (e.g., 4-4-2 $\to$ 3-5-2) based on game state.
                \item \textbf{Noisy Data:} Players swap positions temporarily (overlap, pressing).
                \item \textbf{The Gap:} Standard methods often assume static formations or are too sensitive to noise.
            \end{itemize}
            
            \vspace{0.5cm}
            
            \textbf{Project Goal}
            \begin{itemize}
                \item Reproduce and extend \textbf{SoccerCPD} \cite{Kim2022SoccerCPD}, a change-point detection framework.
                \item Distinguish \textbf{Strategic Changes} (permanent) from \textbf{Tactical Adjustments} (temporary).
            \end{itemize}
        \end{column}
        \begin{column}{0.4\textwidth}
            \centering
            \includegraphics[width=0.9\linewidth]{../figures/delany_form_change.png}
            \captionof{figure}{Formation shifts captured via Delaunay triangulation.}
        \end{column}
    \end{columns}
\end{frame}

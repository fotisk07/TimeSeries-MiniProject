\subsection{Role Change-Point Detection}
\begin{frame}{RoleCPD: Method & Algorithm}
    \frametitle{RoleCPD: Detecting Intraphase Swaps}
    \textbf{Goal:} Detect permanent swaps within a stable formation.

    \textbf{Metric:} Hamming Distance on permutation sequence $\pi_t$: $$d(\pi_t, \pi_{t'}) = \frac{1}{N} \sum_p \mathbb{I}_{\pi_t(X_p) \neq \pi_{t'}(X_p)}$$
            
    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Recursive Segmentation Algorithm}
            \begin{itemize}
                \item \textbf{Filter:} Ignore high-noise frames (Switch Rate $> 0.7$).
                \item \textbf{Search:} Recursive discrete g-segmentation (graph-based).
                \item \textbf{Stop:} If $p\text{-value} > 0.01$ or segment $< 5$ min.
            \end{itemize}
        \end{column}

\begin{frame}{FormCPD 2}
    \frametitle{FormCPD: Change-Point Detection}

    \begin{columns}
        \begin{column}{0.5\textwidth}
            \textbf{Discrete g-segmentation}
            \begin{itemize}
                \item Sequence $\{A(t)\}$.
                \item Distance $d_M = \|A(t)-A(t')\|_{1,1}$.
                \item Constraints: $p<0.01$, min duration 5 min.
            \end{itemize}
        \end{column}

\subsection{Role Change-Point Detection}
\begin{frame}{RoleCPD}
    \frametitle{RoleCPD: Intraphase Tactical Shifts}

    \begin{columns}
        \begin{column}{0.6\textwidth}
            \begin{itemize}
                \item Detect permanent role swaps.
                \item Switch Rate:
                \[
                d(\pi_t,\pi_{t'}) = \frac{1}{N}
                \sum_p \mathbbm{1}_{\pi_t(X_p)\neq\pi_{t'}(X_p)}
                \]
                \item Recursive g-segmentation after filtering.
            \end{itemize}
        \end{column}
